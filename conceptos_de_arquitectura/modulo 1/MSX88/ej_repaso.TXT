ORG 1000H
TABLA DB "A2B56743P2"
RESUL DB ?

ORG 3000H
VERIFICA: MOV AL, [BX]
MOV DL, 30H
COMPARONUM: CMP AL, DL
JZ FUENUM
INC DL
CMP DL, 3AH ; COMPARO SI SE PASO DE NUMEROS
JZ NO_NUM
JMP COMPARONUM
FUENUM: ADD BX, 10
MOV BYTE PTR [BX], 00H
SUB BX, 10
JMP SIGO
NO_NUM: ADD BX, 10
MOV BYTE PTR [BX], 0FFH
SUB BX, 10
SIGO: INC BX
DEC CL
JNZ VERIFICA
RET

ORG 3500H
CUENTA: MOV DL, [BX]
CMP DL, 00H
JNZ NOESNUM
INC AL
NOESNUM: INC BX
DEC CL
JNZ CUENTA
RET

ORG 2000H
MOV CL, 10
MOV BX, OFFSET TABLA
CALL VERIFICA
MOV CL, 10
MOV BX, OFFSET RESUL
MOV AL, 0
CALL CUENTA
INT 0
END
