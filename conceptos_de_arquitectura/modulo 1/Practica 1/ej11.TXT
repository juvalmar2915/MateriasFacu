ORG 1000H
CADENA DB ?
FIN_CADENA DB 00H
CARACTER DB ?
ORG 3000H
CONCAR: MOV CL, [BX]
CMP CL,AH
JNZ ESDISTINTO
MOV [BX], 58H ;REEMPLAZO POR UNA X EL CARACTER ENCONTRADO
ESDISTINTO: INC BX
CMP CL, AL
JNZ CONCAR
RET 
ORG 2000H
MOV DL, 0
MOV AL, FIN_CADENA
MOV BX, OFFSET CADENA
MOV AH, CARACTER
CALL CONCAR
HLT
END
