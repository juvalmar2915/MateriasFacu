ORG 1000H
CADENA DB ?
VOCALES DB 41H,45H,49H,4FH,55H
ORG 3000H
ES_VOCAL: MOV AH, 00H
MOV BX, OFFSET VOCALES
MOV CL,5
VPORV: MOV DL, [BX]
INC BX
CMP AL, DL
JNZ SIGO
MOV AH, 0FFH
JMP FIN
SIGO: DEC CL
JNZ VPORV
FIN: RET
VOCAL3S: MOV CX, OFFSET VOCALES-OFFSET CADENA
VOCAL3S1: MOV AL, [BX]
PUSH CX
PUSH DX
PUSH BX
CALL ES_VOCAL
POP BX
POP DX
POP CX
INC BX
CMP AH, 0FFH
JNZ NOVOCAL
INC DX
NOVOCAL: DEC CX
JNZ VOCAL3S1
RET
ORG 2000H
MOV DX, 0
MOV BX, OFFSET CADENA
CALL VOCAL3S
HLT
END
