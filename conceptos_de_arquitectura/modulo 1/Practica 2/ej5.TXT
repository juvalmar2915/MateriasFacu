ORG 1000H 
MSJ  DB  "INGRESE UN NUMERO:" 
FIN  DB  ?
NOCARAC DB "CARACTER NO VALIDO"
FIN2 DB ?
ORG 1500H 
NUM  DB  ?

ORG 3000H
ES_NUM: PUSH CX
MOV AX, [BX]
MOV CL, 30H ; REPRESENTA EL 0 EN HEXADECIMAL
SIGO: CMP AL, CL
JZ FUENUM
INC CL
CMP CL, 3AH ; REPRESENTA LO QUE SIGUE AL 9 EN HEXADECIMAL (":")
JNZ SIGO
MOV AH, 00H
JMP NOFUENUM
FUENUM: MOV AH, 0FFH
NOFUENUM: POP CX
RET

ORG 2000H
MOV BX, OFFSET MSJ  
MOV AL, OFFSET FIN-OFFSET MSJ
INT 7  
MOV BX, OFFSET NUM
INT 6
CALL ES_NUM
CMP AH, 00H
JNZ NOIMP
MOV AL, OFFSET FIN2-OFFSET NOCARAC
MOV BX, OFFSET NOCARAC
INT 7
NOIMP: MOV AL, 1  
INT 7  
MOV CL, NUM  
INT 0
END
