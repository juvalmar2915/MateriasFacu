CERO DB   "CERO  "
UNO DB    "UNO   "
DOS DB    "DOS   "
TRES DB   "TRES  "
CUATRO DB "CUATRO"
CINCO DB  "CINCO "
SEIS DB   "SEIS  "
SIETE DB  "SIETE "

NUEVE DB  "NUEVE "

ORG 3000H
QUENUMES: PUSH AX
PUSH CX
MOV CX, [BX]
MOV AL, 30H
MOV BX, OFFSET CERO ;MARCO DONDE COMIENZAN LAS LETRAS ESCRITAS
FOR9: CMP AL, CL
JZ FIN
ADD BX, 6 ; AVANZO AL SIGUIENTE NUMERO EN LETRAS
INC AL
CMP AL, 3AH
JNZ FOR9 
FIN: POP CX
POP AX
RET

ORG 2000H
MOV CL, 00H
SIGO1: CMP CL, 01H ; VERIFICO SI YA SE ANOTO EL 0 UNA VEZ PARA DECREMENTARLO EN CASO DE SER ASI
JNZ SIGO2
DEC CL
SIGO2: MOV BX, OFFSET IMP
MOV AL, OFFSET NUM-OFFSET IMP
INT 7
MOV BX, OFFSET NUM
INT 6
CALL QUENUMES
MOV AL, 6
INT 7
MOV BX, OFFSET NUM
MOV DX, [BX]
CMP DL, 30H
JNZ SIGO1
INC CL
CMP CL, 02H
JNZ SIGO2
INT 0
END
