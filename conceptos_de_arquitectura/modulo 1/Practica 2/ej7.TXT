ORG 1000H 
MSJ  DB  "INGRESE UN NUMERO: " 
FIN1  DB  ?
MSJ2 DB "INGRESE OTRO NUMERO: "
FIN2 DB ?
ORG 1500H 
NUM  DB  ?
NUM2 DB ?
RESUL DB "0?"
FINRESUL DB ?

ORG 3000H
QUENUMES: PUSH AX   ; PARA MAYOR FACILIDAD SE PODRIA HABER RESTADO 30H DIRECTAMENTE Y GUARDARLO EN UN REGISTRO
PUSH CX
MOV CX, [BX]
MOV AL, 30H
MOV DL, 00H ; VOY A DEVOLVER EL NUMERO REPRESENTADO POR REGISTRO POR LO QUE NO HAGO UN PUSH DEL REGISTRO DX
FOR9: CMP AL, CL
JZ FIN
INC DL ; REPRESENTO EL DIGITO EN HEXADECIMAL REPRESENTANDO EL 00H EL 0 HASTA EL 09H EL 9
INC AL
CMP AL, 3AH
JNZ FOR9 
FIN: POP CX
POP AX
RET

ORG 2000H
MOV BX, OFFSET MSJ
MOV AL, OFFSET FIN1-OFFSET MSJ
INT 7
MOV BX, OFFSET NUM
INT 6
CALL QUENUMES
MOV CL, DL ; GUARDO EN CL EL DIGITO QUE REPRESENTA EL PRIMER NUMERO
MOV BX, OFFSET MSJ2
MOV AL, OFFSET FIN2-OFFSET MSJ2
INT 7
MOV BX, OFFSET NUM2
INT 6
CALL QUENUMES
MOV BX, OFFSET RESUL
ADD CL, DL ; EFECTUO LA SUMA DE NUMEROS
CMP CL, 0AH ; COMPARO MI NUMERO PARA VER SI FUE MAYOR QUE 10
JS DESTRANFORMO ; SI ES SIGNO SIGNIFICA QUE ES MENOR POR LO QUE NO HAY QUE TOCAR DICHO NUMERO
MOV BYTE PTR [BX], 31H ; VOY A TENER QUE IMPRIMIR UN 1 DELANTE DEL NUMERO QUE DIO LA SUMA
SUB CL, 0AH ; RESTO 10 PARA PODER ALMACENAR EL MENOR NUMERO 
DESTRANFORMO: INC BX ; PASO A LA POSICION DEL SIGNO DE PREGUNTA (LO REALIZO SIEMPRE YA QUE COMIENZA EN LA POS DEL 0)
ADD CL, 30H ; SUMO A CL LA CANTIDAD PARA TRANSFORMARLO EN UN NUMERO SEGUN EL CODIGO ASCII
MOV [BX], CL
MOV AL, OFFSET FINRESUL-OFFSET RESUL
MOV BX, OFFSET RESUL
INT 7
INT 0
END
