ORG 1000H 
MSJ  DB  "INGRESE UN NUMERO: " 
FIN1  DB  ?
MSJ2 DB "INGRESE OTRO NUMERO: "
FIN2 DB ?
ORG 1500H 
NUM  DB  ?
NUM2 DB ?
RESUL DB "0?"

ORG 3000H
QUENUMES: PUSH AX ; PARA MAYOR FACILIDAD SE PODRIA HABER RESTADO 30H EN VEZ DE TODO EL PROCESO
PUSH CX
MOV CX, [BX]
MOV AL, 30H
MOV DL, 00H ; VOY A DEVOLVER EL NUMERO REPRESENTADO POR REGISTRO POR LO QUE NO HAGO UN PUSH DEL REGISTRO DX
FOR9: CMP AL, CL
JZ FIN
INC DL ; REPRESENTO EL DIGITO EN HEXADECIMAL REPRESENTANDO EL 00H EL 0 HASTA EL 09H EL 9
INC AL
CMP AL, 3AH
JNZ FOR9 
FIN: POP CX
POP AX
RET

ORG 2000H
MOV BX, OFFSET MSJ
MOV AL, OFFSET FIN1-OFFSET MSJ
INT 7
MOV BX, OFFSET NUM
INT 6
CALL QUENUMES
MOV CL, DL ; GUARDO EN CL EL DIGITO QUE REPRESENTA EL PRIMER NUMERO
MOV BX, OFFSET MSJ2
MOV AL, OFFSET FIN2-OFFSET MSJ2
INT 7
MOV BX, OFFSET NUM2
INT 6
CALL QUENUMES
MOV BX, OFFSET RESUL
SUB CL, DL ; RESTO AL PRIMER NUMERO EL SEGUNDO
JNS NOMODIFICO ; SI SE ACTIVO EL SIGNO TENGO QUE AGREGAR UN MENOS ADELANTE DEL RESULTADO
ADD CL, DL ; RETORNO AL NUMERO QUE TENIA GUARDADO EN CL Y HAGO LA RESTA AL REVES
SUB DL, CL
MOV CL, DL ; DEJO GUARDADO EL RESULTADO EN CL
MOV BYTE PTR [BX], 2DH ; REEMPLAZO EL 0 POR UN SIGNO NEGATIVO
NOMODIFICO: INC BX
ADD CL, 30H
MOV [BX], CL
MOV AL, 2
MOV BX, OFFSET RESUL
INT 7
INT 0
END
