.data
TABLA: .word 4,11,18,6,17,28,9,0,11,23,15,6,37,29,14
MENOR: .word 20
MAYOR: .word 10
CANTIDAD: .word 15
TOTAL: .word 0
NUEVO: .word 0

.code
ld $t0, CANTIDAD($0)
ld $s0, MENOR($0)
ld $s1, MAYOR($0)
daddi $t1, $0, 0
daddi $s2, $0, 0
daddi $s3, $0, 0
loop: ld $t2, TABLA($t1)
slt $t3, $t2, $s0
beqz $t3, sigo
slt $t3, $s1, $t2
beqz $t3, sigo
daddi $s2, $s2, 1
sd $t2, NUEVO($s3)
daddi $s3, $s3, 8
sigo: daddi $t0, $t0, -1
daddi $t1, $t1, 8
bnez $t0, loop
sd $s2, CANTIDAD($0)
halt
