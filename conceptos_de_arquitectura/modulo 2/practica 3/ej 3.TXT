.data
CONTROL: .word32 0x10000
DATA: .word32 0x10008
.code
lwu $s6, CONTROL($zero)
lwu $s7, DATA($zero)
jal ingreso
dadd $a0, $0, $v0
jal ingreso
dadd $a1, $0, $v0
jal resultado
halt

ingreso: daddi $t1, $zero, 9 ; entrada de mensaje
daddi $t2, $zero, 10
loop: sd $t1, 0($s6)
ld $v0, 0($s7)
daddi $v0, $v0, -48
slt $t4, $v0, $t2
beqz $t4, loop
daddi $t0, $0, -1
slt $t4, $t0, $v0
beqz $t4, loop
jr $ra

resultado: daddi $t0, $0, 1
dadd $v0, $a0, $a1
sd $v0, 0($s7)
sd $t0, 0($s6)
jr $ra
